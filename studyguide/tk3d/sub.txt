        const script = document.createElement("script");
        script.defer = true;
        script.src = src;
        script.type = "text/javascript";
        if (onLoad) {
          script.onload = onLoad;
        }
        script.onerror = onError;
        document.head.append(script);
      };

          s.src.match(
            "^https?://*main.js$"
          )
        )

          tanks3dLoadScriptAsync(
